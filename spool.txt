    function getPageData() {
        const search = (document.getElementById('searchSerial').value || "").toLowerCase();
        const fModel = document.getElementById('filterModel').value;
        const fCRM = document.getElementById('filterCRM').value;
        
        let filtered = window.localDB.filter(i => {
            const wire = i.wireType || (i.model && i.model.includes('У') ? 'Favero' : 'Китайский');
            
            const matchesSearch = (i.serial.toLowerCase().includes(search) || (i.notes||"").toLowerCase().includes(search));
            const matchesModel = fModel === "" || wire === fModel;
            let matchesCRM = true;
            if (fCRM === '1') matchesCRM = !!i.isInCRM;
            if (fCRM === '0') matchesCRM = !i.isInCRM;
            return matchesSearch && matchesModel && matchesCRM;
        });

        filtered.sort((a, b) => {
            let valA, valB;
            if (currentSort.field === 'serial') { valA = a.serial.toLowerCase(); valB = b.serial.toLowerCase(); } 
            else if (currentSort.field === 'wireType') { 
                valA = (a.wireType || a.model || "").toLowerCase(); 
                valB = (b.wireType || b.model || "").toLowerCase(); 
            } 
            else if (currentSort.field === 'spoolModel') {
                valA = a.spoolModel || "0";
                valB = b.spoolModel || "0";
            }
            else { valA = a.globalSeq || 0; valB = b.globalSeq || 0; }
            if (valA < valB) return currentSort.direction === 'asc' ? -1 : 1;
            if (valA > valB) return currentSort.direction === 'asc' ? 1 : -1;
            return 0;
        });

        const totalPages = Math.ceil(filtered.length / itemsPerPage) || 1;
        if(currentPage > totalPages) currentPage = 1;
        const start = (currentPage - 1) * itemsPerPage;
        return filtered.slice(start, start + itemsPerPage);
    }

    function addSwipeListeners() {
        if(window.innerWidth > 991) return; 
        const rows = document.querySelectorAll('.table tr');
        
        rows.forEach(row => {
            let startX, currentX;
            const content = row.querySelectorAll('.mobile-row-content'); 
            
            row.addEventListener('touchstart', (e) => {
                startX = e.touches[0].clientX;
            }, {passive: true});
            
            row.addEventListener('touchmove', (e) => {
                currentX = e.touches[0].clientX;
                const diff = currentX - startX;
                if(diff < 0 && diff > -160) {
                    content.forEach(el => el.style.transform = `translateX(${diff}px)`);
                }
            }, {passive: true});
            
            row.addEventListener('touchend', (e) => {
                const diff = currentX - startX;
                if(diff < -70) { 
                    content.forEach(el => el.style.transform = `translateX(-140px)`);
                } else {
                    content.forEach(el => el.style.transform = `translateX(0)`);
                }
            });
        });
    }

    function updateSmartSerial() {
        if (!isAutoSerialEnabled) return; 

        const wire = document.getElementById('wireType').value;
        const prefix = document.getElementById('serialPrefix').value;
        const seq = document.getElementById('globalSeq').value;

        let effectivePrefix = prefix;
        
        if (wire === "Favero") {
            const lastDotIndex = effectivePrefix.lastIndexOf('.');
            if (lastDotIndex !== -1) {
                effectivePrefix = effectivePrefix.slice(0, lastDotIndex) + 'У' + effectivePrefix.slice(lastDotIndex);
            } else {
                effectivePrefix += 'У';
            }
        }

        const serial = `${effectivePrefix}${seq}`;
        document.getElementById('serialNumber').value = serial;
    }

    async function executeBulkAddAsync() {
        const count = parseInt(document.getElementById('baCount').value) || 1;
        const prefixRaw = document.getElementById('baPrefix').value.trim();
        const startNum = parseInt(document.getElementById('baStartNum').value);
        const wire = document.getElementById('baWireType').value;
        const sModel = document.getElementById('baSpoolModel').value;
        const bMonth = parseInt(document.getElementById('baMonth').value);
        const bYear = parseInt(document.getElementById('baYear').value);
        
        let startSeq = (!isNaN(startNum)) ? startNum : 1;

        // --- DUPLICATE CHECK FOR ENTIRE RANGE ---
        for (let j = 0; j < count; j++) {
            const checkSeq = startSeq + j;
            if (window.localDB.some(x => x.globalSeq === checkSeq)) {
                showToast(`Ошибка: Номер #${checkSeq} уже занят. Выберите другой диапазон.`, 'error');
                return;
            }
        }
        
        // Show Progress UI
        const container = document.getElementById('baProgressContainer');
        const bar = document.getElementById('baProgressBar');
        const text = document.getElementById('baProgressText');
        const submitBtn = document.getElementById('baSubmitBtn');
        const cancelBtn = document.getElementById('baCancelBtn');

        container.style.display = 'block';
        submitBtn.disabled = true;
        cancelBtn.disabled = true;

        let currentSeq = startSeq;
        let createdCount = 0;

        for(let i=0; i < count; i++) {
            let effectivePrefix = prefixRaw;
            if (wire === "Favero") {
                const lastDotIndex = effectivePrefix.lastIndexOf('.');
                if (lastDotIndex !== -1) {
                    effectivePrefix = effectivePrefix.slice(0, lastDotIndex) + 'У' + effectivePrefix.slice(lastDotIndex);
                } else {
                    effectivePrefix += 'У';
                }
            }
            
            let serial = `${effectivePrefix}${currentSeq}`;
            const record = { 
                serial: serial, 
                globalSeq: currentSeq, 
                prefix: prefixRaw,
                wireType: wire,
                spoolModel: sModel,
                month: bMonth,
                year: bYear,
                notes: "Массовое добавление", 
                isInCRM: false, 
                createdAt: serverTimestamp() 
            };
            
            await addDoc(collection(dbRef, "spools"), record);
            
            createdCount++;
            currentSeq++;
            
            // Update progress UI
            const percent = Math.round((createdCount / count) * 100);
            bar.style.width = `${percent}%`;
            text.innerText = `${percent}%`;
        }
        
        setTimeout(() => {
            bootstrap.Modal.getInstance(document.getElementById('bulkAddModal')).hide();
            showToast(`Добавлено: ${count} шт.`);
        }, 300);
    }

    function handleShipmentKey(e) {
        if (e.key === 'Enter') {
            e.preventDefault();
            const query = e.target.value.trim().toLowerCase();
            if (!query) return;

            // 1. Try exact serial match
            let bestMatch = window.localDB.find(item => item.serial.toLowerCase() === query);
            
            // 2. Try exact sequence number match
            if (!bestMatch) {
                bestMatch = window.localDB.find(item => String(item.globalSeq) === query);
            }

            // 3. Fallback to first visible result from the search list
            if (!bestMatch) {
                const matches = window.localDB.filter(item => {
                    const isInBuffer = window.shipmentBuffer.some(b => b.id === item.id);
                    const matchesQuery = item.serial.toLowerCase().includes(query) || (item.globalSeq && String(item.globalSeq).includes(query));
                    return !isInBuffer && matchesQuery;
                });
                if (matches.length > 0) bestMatch = matches[0];
            }

            if (bestMatch) {
                // Check if already in buffer
                if (window.shipmentBuffer.some(b => b.id === bestMatch.id)) {
                    showToast('Уже в списке', 'warning');
                } else {
                    window.addToShipment(bestMatch.id);
                }
            } else {
                showToast('Катушка не найдена', 'error');
            }
            
            // Re-focus and clear
            e.target.value = '';
            window.handleShipmentSearch('');
            e.target.focus();
        }
    }
    window.handleShipmentKey = handleShipmentKey;

    function handleShipmentSearch(query) {
        const resultsDiv = document.getElementById('shipmentSearchResults');
        if (!query || query.trim() === '') {
            resultsDiv.classList.add('d-none');
            return;
        }
        query = query.toLowerCase();
        
        const matches = window.localDB.filter(item => {
            const isInBuffer = window.shipmentBuffer.some(b => b.id === item.id);
            const matchesQuery = item.serial.toLowerCase().includes(query) || (item.globalSeq && String(item.globalSeq).includes(query));
            return !isInBuffer && matchesQuery;
        }).slice(0, 10);

        let html = '';
        if (matches.length === 0) {
            resultsDiv.classList.remove('d-none');
            resultsDiv.innerHTML = '<div class="p-3 text-muted small text-center">Ничего не найдено</div>';
        } else {
            matches.forEach(item => {
                const wire = item.wireType || (item.model && item.model.includes('У') ? 'Favero' : 'Китайский');
                html += `
                <div class="shipment-item" style="cursor:pointer;" onclick="window.addToShipment('${item.id}')">
                    <div>
                        <span class="fw-bold text-primary">${item.serial}</span>
                        <span class="text-muted small ms-2">${wire}</span>
                    </div>
                    <i class="fas fa-plus text-success"></i>
                </div>`;
            });
            resultsDiv.innerHTML = html;
            resultsDiv.classList.remove('d-none');
        }
    }
    window.handleShipmentSearch = handleShipmentSearch;

    function renderShipmentBuffer() {
        const list = document.getElementById('shipmentBufferList');
        const countSpan = document.getElementById('shipmentCount');
        const summaryDiv = document.getElementById('shipmentSummary');
        
        if (countSpan) countSpan.innerText = window.shipmentBuffer.length;
        
        if (window.shipmentBuffer.length === 0) {
            if(summaryDiv) summaryDiv.innerHTML = '';
            list.innerHTML = `
                <div class="text-center text-muted py-5 mt-3">
                    <i class="fas fa-box-open fa-3x mb-3 opacity-50"></i>
                    <p class="mb-1">Буфер пуст</p>
                </div>`;
            return;
        }

        // Calculate Grouping Statistics
        const stats = window.shipmentBuffer.reduce((acc, item) => {
            const wire = item.wireType || (item.model && item.model.includes('У') ? 'Favero' : 'Китайский');
            acc[wire] = (acc[wire] || 0) + 1;
            return acc;
        }, {});

        if(summaryDiv) {
            summaryDiv.innerHTML = Object.entries(stats).map(([type, count]) => {
                const color = type === 'Favero' ? 'secondary' : 'primary';
                return `<span class="badge badge-soft-${color} border">${type}: ${count}</span>`;
            }).join('');
        }

        let html = '';
        [...window.shipmentBuffer].reverse().forEach(item => {
            const wire = item.wireType || (item.model && item.model.includes('У') ? 'Favero' : 'Китайский');
            html += `
            <div class="card p-2 border shadow-sm">
                <div class="d-flex justify-content-between align-items-center">
                    <div>
                        <div class="fw-bold text-primary">${item.serial}</div>
                        <div class="small text-muted">${wire}</div>
                    </div>
                    <div class="d-flex gap-2">
                        <button class="btn btn-sm btn-outline-danger border-0" onclick="window.removeFromShipment('${item.id}')" title="Убрать">
                            <i class="fas fa-times"></i>
                        </button>
                    </div>
                </div>
            </div>`;
        });
        list.innerHTML = html;
    }

    function saveRecord() {
        const editId = document.getElementById('editId').value;
        const serial = document.getElementById('serialNumber').value.trim();
        const seq = Number(document.getElementById('globalSeq').value);

        if (!serial) { showToast('Введите серийный номер', 'error'); return; }
        
        // --- DUPLICATE CHECKS ---
        if (window.localDB.some(x => x.serial === serial && x.id !== editId)) { 
            showToast('Серийный номер уже существует!', 'error'); 
            return; 
        }
        if (window.localDB.some(x => x.globalSeq === seq && x.id !== editId)) { 
            showToast(`Сквозной номер #${seq} уже занят!`, 'error'); 
            return; 
        }

        const record = {
            serial: serial, 
            prefix: document.getElementById('serialPrefix').value,
            globalSeq: seq, 
            wireType: document.getElementById('wireType').value,
            spoolModel: document.getElementById('spoolModel').value,
            month: Number(document.getElementById('prodMonth').value),
            year: Number(document.getElementById('prodYear').value),
            notes: document.getElementById('notes').value, 
            isInCRM: document.getElementById('isInCRM').checked, 
            updatedAt: serverTimestamp()
        };
        
        if(editId) { 
            updateDoc(doc(dbRef, "spools", editId), record).then(() => showToast('Обновлено')).catch(e => showToast('Ошибка: ' + e.message, 'error'));
        } else { 
            record.createdAt = serverTimestamp(); 
            addDoc(collection(dbRef, "spools"), record).then(() => showToast('Создано')).catch(e => showToast('Ошибка: ' + e.message, 'error'));
        }
        offcanvasInstance.hide();
    }
    window.saveRecord = saveRecord;

            });
        } else {
            deleteDoc(doc(dbRef, "spools", recordToDeleteId)).then(() => showToast('Удалено'));
        }
    }
    window.executeDelete = executeDelete;

    function updateCharts() {
        const isDark = document.documentElement.getAttribute('data-theme') === 'dark';
        const tickColor = isDark ? '#94a3b8' : '#64748b';
        const ctxM = document.getElementById('modelPieChart');
        if(ctxM) {
            if(modelPieInstance) modelPieInstance.destroy();
            const counts = [ 
                window.localDB.filter(x=>x.wireType === 'Китайский' || (x.model && !x.model.includes('У') && !x.wireType)).length, 
                window.localDB.filter(x=>x.wireType === 'Favero' || (x.model && x.model.includes('У'))).length 
            ];
            modelPieInstance = new Chart(ctxM, {
                type: 'doughnut',
                data: { labels: ['Китайский','Favero'], datasets: [{ data: counts, backgroundColor: ['#0d6efd','#6f42c1'], borderWidth:0 }] },
                options: { responsive: true, maintainAspectRatio: false, cutout: '70%', plugins: { legend: {position:'bottom', labels: {boxWidth: 10, color: tickColor}} } }
            });
        }
    }
    window.updateCharts = updateCharts;

    function toggleTheme() {
        const h = document.documentElement; const next = h.getAttribute('data-theme') === 'dark' ? 'light' : 'dark';
        h.setAttribute('data-theme', next); localStorage.setItem('theme', next);
        const icon = document.getElementById('themeIcon');
        if (icon) icon.className = next==='dark' ? 'fas fa-sun' : 'fas fa-moon';
        updateCharts();
    }
    window.toggleTheme = toggleTheme;

    function clearLocalData() { localStorage.clear(); location.reload(); }
    window.clearLocalData = clearLocalData;

    const savedTheme = localStorage.getItem('theme') || 'light'; document.documentElement.setAttribute('data-theme', savedTheme);

    initApp();
</script>
